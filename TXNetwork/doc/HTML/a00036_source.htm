481 bool TX_SOCKET :: _txInitAsServer (const char*, u_short port, sock_type blocking, bool useTCP)
482  {
483  addrinfo hints = {};
484  hints.ai_family = AF_INET;
485  hints.ai_socktype = (useTCP? SOCK_STREAM : SOCK_DGRAM);
486  hints.ai_flags = AI_PASSIVE;
487 
488  addrinfo *servinfo = NULL;
489 
490  int _port_ = port;
491 
492  char str[6] = "";
493 
494  sprintf (str, "%d", _port_);
495 
496  int err = getaddrinfo (NULL, str, &hints, &servinfo);
497  if (err == -1) return false;
498 
499  _sock = 0;
500  for (addrinfo* p = servinfo; p; p = p->ai_next)
501  {
502  _sock = socket (p->ai_family, p->ai_socktype, p->ai_protocol);
503 
504  if ( (int) _sock == -1) continue;
505 
506  if (bind (_sock, p->ai_addr, int (p->ai_addrlen)) == -1)
507  {
508  closesocket (_sock);
509  continue;
510  }
511 
512  break;
513  }
514 
515  if (!_sock) return false;
516 
517  freeaddrinfo (servinfo);
518 
519  if (useTCP)
520  {
521  if (listen (_sock, 0x10)) return false;
522 
523  int client_addr_size = sizeof (_client_addr);
524 
525  _client_sock_for_serv = accept (_sock, (sockaddr* ) &_client_addr, &client_addr_size);
526 
527  if (_client_sock_for_serv == 0) return false;
528  }
529 
530  if (blocking == TX_NONBLOCK)
531  {
532  u_long nonblocking = 1;
533  if (ioctlsocket (_client_sock_for_serv, FIONBIO, &nonblocking) == SOCKET_ERROR) return false;
534  }
535 
536  return true;
537  }

539 bool TX_SOCKET :: _txInitAsClient (const char* IP, u_short port, sock_type blocking, bool useTCP)
540  {
541  hostent* he = gethostbyname (IP);
542 
543  _sock = useTCP? socket (AF_INET, SOCK_STREAM, IPPROTO_IP) :
544  socket (AF_INET, SOCK_DGRAM, _broadcats? 0 : IPPROTO_UDP);
545 
546  if (_sock == INVALID_SOCKET) return false;
547 
548  char broadcast = _broadcats;
549  if (_broadcats) if (setsockopt (_sock, SOL_SOCKET, SO_BROADCAST, &broadcast, sizeof (broadcast)) == -1)
550  return perror ("setsockopt (SO_BROADCAST)"), 1;
551 
552  _sock_addr = {};
553 
554  _sock_addr.sin_family = AF_INET;
555  _sock_addr.sin_port = htons (port);
556  _sock_addr.sin_addr = * (in_addr*) he->h_addr;
557 
558  if (useTCP)
559  if (connect (_sock, (sockaddr* ) &_sock_addr, sizeof (_sock_addr)) != 0) return false;
560 
561  if (blocking == TX_NONBLOCK)
562  {
563  u_long nonblocking = 1;
564  if (ioctlsocket (_sock, FIONBIO, &nonblocking) == SOCKET_ERROR) return false;
565  }
566 
567  return true;
568  }

579  return answ;

583  {
584  switch (to._type)

587  case TX_CLIENT: answ = send (to._sock, (const char*) buf, size_buf, 0); break;
588 
589  case TX_BLOCK:
590  case TX_NONBLOCK: return TXN_ERROR;
591 
592  case TX_NOTYPE: return 0;

595  }
596  }
