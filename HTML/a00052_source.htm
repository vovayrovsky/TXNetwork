433 bool TX_SOCKET :: _txInitHowServer(__attribute__((unused)) const char* IP, u_short port, sock_type type_, bool TCP)
434 #endif
435 {
436 addrinfo hints = {};
437 hints.ai_family = AF_INET;
438 hints.ai_socktype = TCP? SOCK_STREAM : SOCK_DGRAM;
439 hints.ai_flags = AI_PASSIVE;
440 
441 addrinfo *servinfo = NULL;
442 
443 int _port_ = port;
444 
445 char str[6] = "";
446 
447 sprintf(str, "%d", _port_);
448 
449 int err = getaddrinfo (NULL, str, &hints, &servinfo);
450 if (err == -1) return false;
451 
452 _sock = 0;
453 for (addrinfo* p = servinfo; p; p = p->ai_next)
454  {
455  _sock = socket (p->ai_family, p->ai_socktype, p->ai_protocol);
456 
457  if ((int) _sock == -1)
458  continue;
459 
460  if (bind (_sock, p->ai_addr, p->ai_addrlen) == -1)
461  {
462  closesocket(_sock);
463  continue;
464  }
465 
466  break;
467  }
468 
469 if (!_sock)
470  return false;
471 
472 freeaddrinfo (servinfo);
473 
474 if (TCP)
475  {
476  if (listen(_sock, 0x10))
477  return false;
478 
479  int client_addr_size = sizeof(_client_addr);
480 
481  _client_sock_for_serv = accept(_sock, (sockaddr* ) &_client_addr, &client_addr_size);
482 
483  if (_client_sock_for_serv == 0) return false;
484  }
485 
486 if (type_ == 4)
487  {
488  u_long mode = 1; //Socket is non-blocked;
489 
490  if (ioctlsocket(_client_sock_for_serv, FIONBIO, &mode) == SOCKET_ERROR)
491  {
492  return false;
493  }
494  }
495 return true;
496 }

498 bool TX_SOCKET :: _txInitHowClient(const char* IP, u_short port, sock_type type_, bool TCP)
499 {
500 hostent* he = gethostbyname (IP);
501 _sock = TCP? socket(AF_INET, SOCK_STREAM, IPPROTO_IP) : socket(AF_INET, SOCK_DGRAM, _broadcats? 0 : IPPROTO_UDP);
502 
503 if (_sock == INVALID_SOCKET)
504  return false;
505 
506 char broadcast = _broadcats;
507 if (_broadcats) if (setsockopt (_sock, SOL_SOCKET, SO_BROADCAST, &broadcast, sizeof (broadcast)) == -1)
508  return perror ("setsockopt (SO_BROADCAST)"), 1;
509 
510 _sock_addr = {};
511 
512 _sock_addr.sin_family = AF_INET;
513 _sock_addr.sin_port = htons(port);
514 _sock_addr.sin_addr = *(in_addr*) he->h_addr;
515 
516 if (TCP)
517  {
518  if (connect(_sock, (sockaddr* ) &_sock_addr, sizeof(_sock_addr)) != 0)
519  return false;
520  }
521 
522 if (type_ == 4)
523  {
524  u_long mode = 1; //Socket is non-blocked;
525 
526  if (ioctlsocket(_sock, FIONBIO, &mode) == SOCKET_ERROR)
527  {
528  return false;
529  }
530  }
531 
532 return true;
533 }
