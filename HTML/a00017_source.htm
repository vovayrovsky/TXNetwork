483 bool TX_SOCKET :: _txInitAsServer (const char*, u_short port, sock_type blocking, bool useTCP)
484  {
485  addrinfo hints = {};
486  hints.ai_family = AF_INET;
487  hints.ai_socktype = (useTCP? SOCK_STREAM : SOCK_DGRAM);
488  hints.ai_flags = AI_PASSIVE;
489 
490  addrinfo *servinfo = NULL;
491 
492  int _port_ = port;
493 
494  char str[6] = "";
495 
496  sprintf (str, "%d", _port_);
497 
498  int err = getaddrinfo (NULL, str, &hints, &servinfo);
499  if (err == -1) return false;
500 
501  _sock = 0;
502  for (addrinfo* p = servinfo; p; p = p->ai_next)
503  {
504  _sock = socket (p->ai_family, p->ai_socktype, p->ai_protocol);
505 
506  if ( (int) _sock == -1) continue;
507 
508  if (bind (_sock, p->ai_addr, int (p->ai_addrlen)) == -1)
509  {
510  closesocket (_sock);
511  continue;
512  }
513 
514  break;
515  }
516 
517  if (!_sock) return false;
518 
519  freeaddrinfo (servinfo);
520 
521  if (useTCP)
522  {
523  if (listen (_sock, 0x10)) return false;
524 
525  int client_addr_size = sizeof (_client_addr);
526 
527  _client_sock_for_serv = accept (_sock, (sockaddr* ) &_client_addr, &client_addr_size);
528 
529  if (_client_sock_for_serv == 0) return false;
530  }
531 
532  if (blocking == TX_NONBLOCK)
533  {
534  u_long nonblocking = 1;
535  if (ioctlsocket (_client_sock_for_serv, FIONBIO, &nonblocking) == SOCKET_ERROR) return false;
536  }
537 
538  return true;
539  }

541 bool TX_SOCKET :: _txInitAsClient (const char* IP, u_short port, sock_type blocking, bool useTCP)
542  {
543  hostent* he = gethostbyname (IP);
544 
545  _sock = useTCP? socket (AF_INET, SOCK_STREAM, IPPROTO_IP) :
546  socket (AF_INET, SOCK_DGRAM, _broadcats? 0 : IPPROTO_UDP);
547 
548  if (_sock == INVALID_SOCKET) return false;
549 
550  char broadcast = _broadcats;
551  if (_broadcats) if (setsockopt (_sock, SOL_SOCKET, SO_BROADCAST, &broadcast, sizeof (broadcast)) == -1)
552  return perror ("setsockopt (SO_BROADCAST)"), 1;
553 
554  _sock_addr = {};
555 
556  _sock_addr.sin_family = AF_INET;
557  _sock_addr.sin_port = htons (port);
558  _sock_addr.sin_addr = * (in_addr*) he->h_addr;
559 
560  if (useTCP)
561  if (connect (_sock, (sockaddr* ) &_sock_addr, sizeof (_sock_addr)) != 0) return false;
562 
563  if (blocking == TX_NONBLOCK)
564  {
565  u_long nonblocking = 1;
566  if (ioctlsocket (_sock, FIONBIO, &nonblocking) == SOCKET_ERROR) return false;
567  }
568 
569  return true;
570  }

589  case TX_CLIENT: answ = send (to._sock, (const char*) buf, size_buf, 0); break;
590
