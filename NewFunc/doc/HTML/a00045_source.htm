43 void NetPaint(TX_SOCKET paint_peer)
44 {
45 Netcursor_t cursor = {100, 100, -1, -1, 0};
46 Netcursor_t peer_cursor = {100, 100, -1, -1, 0};
47 
48 while (!GetAsyncKeyState(VK_ESCAPE))
49  {
50  if (peer_cursor.action == -1)
51  break;
52 
53  txSetFillColor (TX_GREEN);
54  txSetColor (TX_GREEN, 10);
55 
56  if (cursor.action == 1) txLine (cursor.old_x, cursor.old_y, cursor.x, cursor.y);
57 
58  txSetFillColor (TX_RED);
59  txSetColor (TX_RED, 10);
60 
61  if (peer_cursor.action == 1) txLine (peer_cursor.old_x, peer_cursor.old_y, peer_cursor.x, peer_cursor.y);
62 
63  cursor.old_x = cursor.x;
64  cursor.old_y = cursor.y;
65 
66  cursor.x = txMouseX();
67  cursor.y = txMouseY();
68 
69  if (txMouseButtons() == 1)
70  cursor.action = 1;
71  else
72  cursor.action = 0;
73 
74  txSendTo(paint_peer, (char*) &cursor , sizeof(cursor));
75  txRecvTo(paint_peer, (char*) &peer_cursor, sizeof(peer_cursor));
76 
77  txSleep(10);
78  }
79 
80 cursor.action = -1;
81 txSendTo(paint_peer, (char*) &cursor, sizeof(cursor));
82 }
