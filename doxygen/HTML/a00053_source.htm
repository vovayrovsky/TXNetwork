423 bool TX_SOCKET :: _txInitHowServer(__attribute__((unused)) const char* IP, u_short port, sock_type type_, bool TCP)
424 #endif
425 {
426 addrinfo hints = {};
427 hints.ai_family = AF_INET;
428 hints.ai_socktype = TCP? SOCK_STREAM : SOCK_DGRAM;
429 hints.ai_flags = AI_PASSIVE;
430 
431 addrinfo *servinfo = NULL;
432 
433 int _port_ = port;
434 
435 char str[6] = "";
436 
437 sprintf(str, "%d", _port_);
438 
439 int err = getaddrinfo (NULL, str, &hints, &servinfo);
440 if (err == -1) return false;
441 
442 _sock = 0;
443 for (addrinfo* p = servinfo; p; p = p->ai_next)
444  {
445  _sock = socket (p->ai_family, p->ai_socktype, p->ai_protocol);
446 
447  if ((int) _sock == -1)
448  continue;
449 
450  if (bind (_sock, p->ai_addr, p->ai_addrlen) == -1)
451  {
452  closesocket(_sock);
453  continue;
454  }
455 
456  break;
457  }
458 
459 if (!_sock)
460  return false;
461 
462 freeaddrinfo (servinfo);
463 
464 if (TCP)
465  {
466  if (listen(_sock, 0x10))
467  return false;
468 
469  int client_addr_size = sizeof(_client_addr);
470 
471  _client_sock_for_serv = accept(_sock, (sockaddr* ) &_client_addr, &client_addr_size);
472 
473  if (_client_sock_for_serv == 0) return false;
474  }
475 
476 if (type_ == 4)
477  {
478  u_long mode = 1; //Socket is non-blocked;
479 
480  if (ioctlsocket(_client_sock_for_serv, FIONBIO, &mode) == SOCKET_ERROR)
481  {
482  return false;
483  }
484  }
485 return true;
486 }

489 {
490 hostent* he = gethostbyname (IP);
491 _sock = TCP? socket(AF_INET, SOCK_STREAM, IPPROTO_IP) : socket(AF_INET, SOCK_DGRAM, _broadcats? 0 : IPPROTO_UDP);
492 
493 if (_sock == INVALID_SOCKET)
494  return false;
495 
496 char broadcast = _broadcats;
497 if (_broadcats) if (setsockopt (_sock, SOL_SOCKET, SO_BROADCAST, &broadcast, sizeof (broadcast)) == -1)
498  return perror ("setsockopt (SO_BROADCAST)"), 1;
499 
500 _sock_addr = {};
501 
502 _sock_addr.sin_family = AF_INET;

506 if (TCP)
